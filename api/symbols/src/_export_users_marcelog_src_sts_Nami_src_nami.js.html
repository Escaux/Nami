 27 var net = require('net');
 28 var events = require('events');
 29 var action = require('./message/action.js');
 30 var namiResponse = require('./message/response.js');
 31 var util = require('util');
 32 var namiEvents = require('./message/event.js');
 33 var timer = require('timers');
 34 
 35 /**
 36  * Nami client.
 37  * @constructor
 38  * @param amiData The configuration for ami.
 39  * @augments EventEmitter
 40  */
 41 function Nami(amiData) {
 42     Nami.super_.call(this);
 43     this.logger = require('log4js').getLogger('Nami.Client');
 44     this.connected = false;
 45     this.amiData = amiData;
 46     this.EOL = "\r\n";
 47     this.EOM = this.EOL + this.EOL;
 48     this.welcomeMessage = "Asterisk Call Manager/1.1" + this.EOL;
 49     this.received = false;
 50     this.responses = { };
 51     this.callbacks = { };
 52 };
 53 // Nami inherits from the EventEmitter so Nami itself can throw events.
 54 util.inherits(Nami, events.EventEmitter);
 55 
 56 /**
 57  * Called when a message arrives and is decoded as an event (namiRawEvent event).
 58  * This will actually instantiate an Event. If the event has an ActionID,
 59  * the corresponding response is looked up and will have this event appended.
 60  * Otherwise, the event "namiEvent" is fired. Also, the event "namiEvent<EventName>"
 61  * is fired (i.e: on event Dial, namiEventDial will be fired).
 62  * 
 63  * @see Nami#onRawMessage(String)
 64  * @param {Event} response An Event message.
 65  * @returns void
 66  */
 67 Nami.prototype.onRawEvent = function (event) {
 68     this.logger.debug('Got event: ' + util.inspect(event));
 69 	if (typeof (event.actionid) !== 'undefined') {
 70 		this.responses[event.actionid].events.push(event);
 71 	}
 72 	if (
 73 		event.event.indexOf('Complete') !== -1
 74 		|| ((typeof(event.eventlist) !== 'undefined') && event.eventlist.indexOf('Complete') != -1)
 75 	) {
 76 		this.callbacks[event.actionid](this.responses[event.actionid]);
 77 	} else {
 78 		this.emit('namiEvent', event);
 79 		this.emit('namiEvent' + event.event, event);
 80 	}
 81 };
 82 
 83 /**
 84  * Called when a message arrives and is decoded as a response (namiRawResponse event).
 85  * This will actually instantiate a Response. If this response has associated events
 86  * (still to be received), it is buffered.
 87  * Otherwise, the callback used in send() will be called with the response.
 88  * @see Nami#onRawMessage(String)
 89  * @see Nami#send(String)
 90  * @param {Response} response A Response message.
 91  * @returns void
 92  */
 93 Nami.prototype.onRawResponse = function (response) {
 94     this.logger.debug('Got response: ' + util.inspect(response));
 95 	if (
 96         (typeof(response.message) !== 'undefined')
 97         && (response.message.indexOf('follow') != -1)
 98     ) {
 99 		this.responses[response.actionid] = response;			
100 	} else if (typeof (this.callbacks[response.actionid]) !== 'undefined') {
101 		this.callbacks[response.actionid](response);
102 	}
103 };
104 
105 /**
106  * Called by onData() whenever a raw message has been read.
107  * Will fire "namiRawEvent" if the raw message represents an event.
108  * Will fire "namiRawResponse" if the raw message represents a response.
109  * @see Nami#onData(String)
110  * @param {String} buffer The raw message read from server.
111  * @returns void
112  */
113 Nami.prototype.onRawMessage = function (buffer) {
114     this.logger.debug('Building raw message: ' + util.inspect(buffer));
115 	if (buffer.indexOf('Event: ') != -1) {
116 		var event = new namiEvents.Event(buffer);
117 		this.emit('namiRawEvent', event);
118 	} else if (buffer.indexOf('Response: ') != -1) {
119 		var response = new namiResponse.Response(buffer);
120 		this.emit('namiRawResponse', response);
121 	} else {
122 		this.logger.warn("Discarded: |" + buffer + "|");
123 	}
124 };
125 
126 /**
127  * Called by node whenever data is available to be read from AMI.
128  * Will fire "namiRawMessage" for every complete message read.
129  * @param {String} data The data read from server.
130  * @see Nami#onRawMessage(String)
131  * @returns void
132  */
133 Nami.prototype.onData = function (data) {
134     this.logger.debug('Got data: ' + util.inspect(data));
135     while ((theEOM = data.indexOf(this.EOM)) != -1) {
136         this.received = this.received.concat(data.substr(0, theEOM));
137         this.emit('namiRawMessage', this.received);
138         this.received = "";
139         data = data.substr(theEOM + this.EOM.length);
140     }
141     this.received = data;
142 };
143 /**
144  * Called when the connection is established to AMI.
145  * @returns void
146  */
147 Nami.prototype.onConnect = function () {
148     this.connected = true;
149 };
150 
151 /**
152  * Called when the first line is received from the server. It will check that
153  * the other peer is a valid AMI server. If not valid, the event "namiInvalidPeer"
154  * will be fired. If not, a login is tried, and onData() is set as the new handler
155  * for incoming data. An anonymous function will handle the login response, firing
156  * "namiLoginIncorrect" if the username/password were not correctly validated.
157  * On successfull connection, "namiConnected" is emitted.
158  * @param {String} data The data read from server.
159  * @see Nami#onData(String)
160  * @see Login(String, String)
161  * @returns void
162  */
163 Nami.prototype.onWelcomeMessage = function (data) {
164     this.logger.debug('Got welcome message: ' + util.inspect(data));
165     var welcome = data.indexOf(this.welcomeMessage);
166 	var self = this;
167     if (welcome == -1) {
168         this.emit('namiInvalidPeer', data);
169     } else {
170         this.socket.on('data', function (data) {
171         	self.onData(data);
172         });
173         this.send(
174         	new action.Login(this.amiData.username, this.amiData.secret),
175         	function (response) {
176         		if (response.response != 'Success') {
177         			self.emit('namiLoginIncorrect');
178         		} else {
179         			self.emit('namiConnected');
180                 }
181         	}
182         );
183     }
184 };
185 /**
186  * Closes the connection to AMI.
187  * @returns void
188  */
189 Nami.prototype.close = function () {
190     this.logger.info('Closing connection');
191     this.removeAllListeners();
192     this.socket.removeAllListeners();
193     this.socket.end();
194 };
195 
196 /**
197  * Opens the connection to AMI.
198  * @returns void
199  */
200 Nami.prototype.open = function () {
201     this.logger.debug('Opening connection');
202     this.socket = new net.Socket();
203     var self = this;
204     this.socket.on('connect', this.onConnect);
205     this.on('namiRawMessage', this.onRawMessage);
206     this.on('namiRawResponse', this.onRawResponse);
207     this.on('namiRawEvent', this.onRawEvent);
208     this.socket.once('data', function (data) {
209     	self.onWelcomeMessage(data); 
210     });
211     this.socket.setEncoding('ascii');
212     this.received = "";
213     this.socket.connect(this.amiData.port, this.amiData.host);
214 };
215 
216 /**
217  * Sends an action to AMI.
218  *
219  * @param {Action} action The action to be sent.
220  * @param {function} callback The optional callback to be invoked when the
221  * responses arrives.
222  *
223  * @returns void
224  */
225 Nami.prototype.send = function (action, callback) {
226     this.logger.debug('Sending: ' + util.inspect(action));
227     this.callbacks[action.ActionID] = callback;
228     this.responses[action.ActionID] = "";
229     this.socket.write(action.marshall());
230 };

232 exports.Nami = Nami;
233 exports.Actions = action;
234 exports.Event = namiEvents;
235 exports.Response = namiResponse;
