 19 var net = require('net');
 20 var events = require('events');
 21 var action = require('./message/action.js');
 22 var namiResponse = require('./message/response.js');
 23 var util = require('util');
 24 var namiEvents = require('./message/event.js');
 25 var timer = require('timers');
 26 
 27 // Constructor
 28 function Nami(amiData) {
 29     Nami.super_.call(this);
 30     this.logger = require('log4js').getLogger('Nami.Client');
 31     this.connected = false;
 32     this.amiData = amiData;
 33     this.EOL = "\r\n";
 34     this.EOM = this.EOL + this.EOL;
 35     this.welcomeMessage = "Asterisk Call Manager/1.1" + this.EOL;
 36     this.received = false;
 37     this.responses = { };
 38     this.callbacks = { };
 39 };
 40 // Nami inherits from the EventEmitter so Nami itself can throw events.
 41 util.inherits(Nami, events.EventEmitter);
 42 
 43 Nami.prototype.onRawEvent = function (event) {
 44     this.logger.debug('Got event: ' + util.inspect(event));
 45 	if (typeof (event.ActionID) !== 'undefined') {
 46 		this.responses[event.ActionID].events.push(event);
 47 	}
 48 	if (
 49 		event.Event.indexOf('Complete') !== -1
 50 		|| ((typeof(event.EventList) !== 'undefined') && event.EventList.indexOf('Complete') != -1)
 51 	) {
 52 		this.callbacks[event.ActionID](this.responses[event.ActionID]);
 53 	} else {
 54 		this.emit('namiEvent', event);
 55 		this.emit('namiEvent' + event.Event, event);
 56 	}
 57 };
 58 
 59 Nami.prototype.onRawResponse = function (response) {
 60     this.logger.debug('Got response: ' + util.inspect(response));
 61 	if (response.Message.indexOf('follow') != -1) {
 62 		this.responses[response.ActionID] = response;			
 63 	} else if (typeof (this.callbacks[response.ActionID]) !== 'undefined') {
 64 		this.callbacks[response.ActionID](response);
 65 	}
 66 };
 67 
 68 Nami.prototype.onRawMessage = function (buffer) {
 69     this.logger.debug('Building raw message: ' + util.inspect(buffer));
 70 	if (buffer.indexOf('Event: ') != -1) {
 71 		var event = new namiEvents.Event(buffer);
 72 		this.emit('namiRawEvent', event);
 73 	} else if (buffer.indexOf('Response: ') != -1) {
 74 		var response = new namiResponse.Response(buffer);
 75 		this.emit('namiRawResponse', response);
 76 	} else {
 77 		this.logger.warn("Discarded: |" + buffer + "|");
 78 	}
 79 };
 80 
 81 Nami.prototype.onData = function (data) {
 82     this.logger.debug('Got data: ' + util.inspect(data));
 83     while ((theEOM = data.indexOf(this.EOM)) != -1) {
 84         this.received = this.received.concat(data.substr(0, theEOM));
 85         this.emit('namiRawMessage', this.received);
 86         this.received = "";
 87         data = data.substr(theEOM + this.EOM.length);
 88     }
 89     this.received = data;
 90 };
 91 Nami.prototype.onConnect = function () {
 92     this.connected = true;
 93 };
 94 
 95 Nami.prototype.onWelcomeMessage = function (data) {
 96     this.logger.debug('Got welcome message: ' + util.inspect(data));
 97     var welcome = data.indexOf(this.welcomeMessage);
 98 	var self = this;
 99     if (welcome == -1) {
100         this.emit('namiInvalidPeer', data);
101     } else {
102         this.socket.on('data', function (data) {
103         	self.onData(data);
104         });
105         this.send(
106         	new action.LoginAction(this.amiData.username, this.amiData.secret),
107         	function (response) {
108         		if (response.Response != 'Success') {
109         			self.emit('namiLoginIncorrect');
110         		}
111         	}
112         );
113     }
114 };
115 Nami.prototype.close = function () {
116     this.logger.info('Closing connection');
117     this.removeAllListeners();
118     this.socket.removeAllListeners();
119     this.socket.end();
120 };
121 
122 Nami.prototype.open = function () {
123     this.logger.debug('Opening connection');
124     this.socket = new net.Socket();
125     var self = this;
126     this.socket.on('connect', this.onConnect);
127     this.on('namiRawMessage', this.onRawMessage);
128     this.on('namiRawResponse', this.onRawResponse);
129     this.on('namiRawEvent', this.onRawEvent);
130     this.socket.once('data', function (data) {
131     	self.onWelcomeMessage(data); 
132     });
133     this.socket.setEncoding('ascii');
134     this.received = "";
135     this.socket.connect(this.amiData.port, this.amiData.host);
136 };
137 
138 Nami.prototype.send = function (action, callback) {
139     this.logger.debug('Sending: ' + util.inspect(action));
140     this.callbacks[action.ActionID] = callback;
141     this.responses[action.ActionID] = "";
142     this.socket.write(action.marshall());
143 };
144 
145 exports.Nami = Nami;
