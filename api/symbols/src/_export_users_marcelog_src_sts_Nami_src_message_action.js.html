 19 message = require('./message.js');
 20 util = require('util');
 21 
 22 function Action(name) {
 23     Action.super_.call(this);
 24     this.id = new Date().getTime();
 25     this.set('ActionID', this.id);
 26     this.set('Action', name);
 27 };
 28 
 29 function LoginAction(username, secret) {
 30     LoginAction.super_.call(this, 'Login');
 31     this.set('Username', username);
 32     this.set('Secret', secret );
 33 };
 34 function CoreShowChannelsAction() {
 35 	CoreShowChannelsAction.super_.call(this, 'CoreShowChannels');
 36 };
 37 
 38 // Inheritance for this module
 39 util.inherits(Action, message.Message);
 40 util.inherits(LoginAction, Action);
 41 util.inherits(CoreShowChannelsAction, Action);
 42 // Exports for this module
 43 exports.LoginAction = LoginAction;
 44 exports.CoreShowChannelsAction = CoreShowChannelsAction;
