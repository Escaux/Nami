 28 /**
 29  * Base action class. Every action sent to AMI must be one of these.
 30  * @constructor
 31  * @param {String} name The name of the action, this is the actual value of the 
 32  * "Action" key in the action message.
 33  * @see Message#marshall(String)
 34  * @augments Message
 35  */
 36 function Action(name) {
 37     Action.super_.call(this);
 38     this.id = new Date().getTime();
 39     this.set('ActionID', this.id);
 40     this.set('Action', name);
 41 };
 42 
 43 /**
 44  * Login Action.
 45  * @constructor
 46  * @param {String} username The username. The value of the "Username" key.
 47  * @param {String} secret The password. The value of the "Secret" key.
 48  * @see Action(String)
 49  * @augments Action
 50  */
 51 function Login(username, secret) {
 52     Login.super_.call(this, 'Login');
 53     this.set('Username', username);
 54     this.set('Secret', secret );
 55 };
