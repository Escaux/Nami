 19 /**

 21  *
 22  * @author Marcelo Gornstein - http://marcelog.github.com

 25 message = require('./message.js');
 26 util = require('util');
 27 
 28 /**
 29  * Base event class. Every async event from the server ends up being an Event()
 30  * @constructor
 31  * @param {String} data A message received from AMI. The End-Of-Message indicator
 32  * has to be already stripped out. This will call unserialize() to build the actual
 33  * Message object.
 34  * @see Message#unmarshall(String)
 35  * @augments Message
 36  */
 37 function Event(data) {
 38     Event.super_.call(this);
 39     this.unmarshall(data);
 40 };
 41 util.inherits(Event, message.Message);
 42 exports.Event = Event;
